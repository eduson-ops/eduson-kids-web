quiz.py · финал 
def load_questions(path):
    questions = []
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            parts = line.strip().split("|")
            if len(parts) != 6:
                continue
            questions.append((parts[0], parts[1:5], int(parts[5])))
    return questions


def ask_question(question):
    text, options, correct = question
    print(f"\n❓ {text}")
    for i, opt in enumerate(options, start=1):
        print(f"   {i}) {opt}")
    try:
        answer = int(input("Твой ответ (1-4): "))
    except ValueError:
        return False
    return answer == correct


def save_result(name, score, total):
    with open("results.txt", "a", encoding="utf-8") as f:
        f.write(f"{name} — {score}/{total}\n")


def main():
    print("🎮 Добро пожаловать в квиз!")
    name = input("Как тебя зовут? ")
    questions = load_questions("questions.txt")
    score = 0
    for q in questions:
        if ask_question(q):
            score += 1
            print("✅ Верно!")
        else:
            print("❌ Мимо.")
    total = len(questions)
    print(f"\n🏆 Результат: {name} — {score}/{total}")
    save_result(name, score, total)
    print("💾 Сохранено в results.txt")


if __name__ == "__main__":
    main()
